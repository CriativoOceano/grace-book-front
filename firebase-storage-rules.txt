# Firebase Storage Rules para Grace Book

# Regras de segurança para Firebase Storage
# Estas regras devem ser aplicadas no Firebase Console

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Permitir leitura pública de todas as imagens
    match /{allPaths=**} {
      allow read: if true;
    }
    
    // Permitir upload apenas para usuários autenticados
    // Para desenvolvimento, você pode usar regras mais permissivas:
    match /{allPaths=**} {
      allow write: if true; // TEMPORÁRIO - Para desenvolvimento
    }
    
    // Para produção, use regras mais restritivas:
    // match /{allPaths=**} {
    //   allow write: if request.auth != null;
    // }
  }
}

# Instruções para aplicar as regras:
# 1. Acesse o Firebase Console: https://console.firebase.google.com/
# 2. Selecione seu projeto: sede-campestre-oceano
# 3. Vá para "Storage" no menu lateral
# 4. Clique na aba "Rules"
# 5. Cole as regras acima
# 6. Clique em "Publish"
